require(rCharts)

grid1 <- rCharts$new()
grid1$field("lib", "libraries/widgets/d3grid")

grid1$set(height = 500, width = 960, container = ".rChart", size = 400, nrow = 8, 
    ncol = 8)
grid1$print()
